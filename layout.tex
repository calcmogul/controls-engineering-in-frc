% Disable automatic indent and provide \indent command
\newlength\tindent
\setlength{\tindent}{\parindent}
\setlength{\parindent}{0pt}
\renewcommand{\indent}{\hspace*{\tindent}}

% Paper header
\pagestyle{fancy}

\renewcommand{\chaptermark}[1]{%
  \markboth{\sffamily\normalsize\bfseries\chaptername\ \thechapter.\ #1}{}}
\renewcommand{\sectionmark}[1]{%
  \markright{\sffamily\normalsize\thesection\hspace{5pt}#1}{}}

% Font setting for the page number in the header
\fancyhf{} \fancyhead[LE,RO]{\sffamily\normalsize\thepage}

\fancyhead[LO]{\rightmark} % Nearest section name on left side of odd pages
\fancyhead[RE]{\leftmark} % Current chapter name on right side of even pages
\renewcommand{\headrulewidth}{0.5pt} % Width of the rule under the header
\addtolength{\headheight}{2.5pt} % Increase spacing around header slightly
\renewcommand{\footrulewidth}{0pt} % Removes the rule in the footer

% Style for when a plain pagestyle is specified
\fancypagestyle{plain}{\fancyhead{}\renewcommand{\headrulewidth}{0pt}}

% Removes the header from odd empty pages at the end of chapters
\setcleardoublepage
